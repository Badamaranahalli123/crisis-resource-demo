<div class="box">
  <label>Severity (1–5):</label><br>
  <input id="severity" type="number" min="1" max="5"><br>

  <label>Population Affected:</label><br>
  <input id="population" type="number"><br>

  <label>Urgency (1–5):</label><br>
  <input id="urgency" type="number" min="1" max="5"><br>

  <button onclick="match()">Recommend Resource</button>

  <h3 id="result"></h3>
</div>

<script>
function match() {
  const s = +severity.value;
  const p = +population.value;
  const u = +urgency.value;

  const score = (s * 3) + (u * 2) + (p / 1000);

  let resource = "General Aid";
  if (score > 20) resource = "Emergency Medical Team";
  else if (score > 12) resource = "Food & Water Supplies";

  document.getElementById("result").innerText =
    "Priority Score: " + score.toFixed(1) +
    " → Recommended Resource: " + resource;
}
</script>
